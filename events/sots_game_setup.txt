### System/Sector Capacity Balancing Removal on Expiration
country_event = {
	id = sots_gameinit.305
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		every_country = {
			limit = {
				has_modifier = legacy_government_balance_v1
				OR = {
					AND = {
						years_passed > 45
						has_trait = "trait_legacy_ringworld_basic"
					}
					AND = {
						years_passed > 15
						has_trait = "trait_legacy_perfect_world"
					}
					AND = {
						years_passed > 5
						has_trait = "trait_legacy_big_world"
					}
				}
			}
			remove_modifier = legacy_government_balance_v1
		}
		every_country = {
			limit = {
				has_modifier = legacy_government_balance_v2
				OR = {
					AND = {
						years_passed > 45
						has_trait = "trait_legacy_ringworld_basic"
					}
					AND = {
						years_passed > 15
						has_trait = "trait_legacy_perfect_world"
					}
					AND = {
						years_passed > 5
						has_trait = "trait_legacy_big_world"
					}
				}
			}
			remove_modifier = legacy_government_balance_v2
		}
		every_country = {
			limit = {
				has_modifier = legacy_government_balance_v1
				NOR = {
					has_government = moral_democracy
					has_government = enlightened_monarchy
				}
			}
			remove_modifier = legacy_government_balance_v1
		}
		every_country = {
			limit = {
				has_modifier = legacy_government_balance_v2
				NOT = {
					has_government = peaceful_bureaucracy
				}
			}
			remove_modifier = legacy_government_balance_v2
		}
	}
}
