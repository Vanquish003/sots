	}

}

empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 1
			OR = {
				has_government = "collective_consciousness"
				has_government = "collective_intelligence"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
		}
