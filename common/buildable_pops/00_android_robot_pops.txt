	}
	
	potential = {
		#always = yes
		#is_species_class = "ROB"
		species_portrait = "robot2"
		has_trait = trait_android
		NOT = { has_technology = tech_robotic_workers }
	}
	
	species = {
		name = "ROBOT_POP_ANDROID"
		class = "ROB"
		portrait = robot2
		traits = {
			trait = "trait_android"
		}
		new_pop_resource_requirement = {
			type = robot_food
			value = 30.0
		}
		pops_auto_growth = 1.0
		pops_auto_upgrade_to = buildable_robot_pop_robot2_1

		pops_can_migrate = yes
		pops_can_reproduce = no
		pops_can_join_factions = yes
		pop_maintenance = 1
		can_generate_leaders = yes

		can_be_modified = yes
